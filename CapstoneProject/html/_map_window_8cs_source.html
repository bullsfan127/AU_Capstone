154  public void Update(GameTime gameTime)
155  {
156  _center.Update(gameTime);
157  _map.Update(gameTime, Vector2.Zero);
158  layerLabel.Update(gameTime);
159  switch (_currentLayer)
160  {
161  case (currentLayer.GROUND):
162  layerLabel.Text = "ground";
163  break;
164  case (currentLayer.MASK):
165  layerLabel.Text = "MASK";
166  break;
167  case (currentLayer.FRINGE):
168  layerLabel.Text = "FRINGE";
169  break;
170  }
171  MouseState mouse = Mouse.GetState();
172  Rectangle mouseColision = new Rectangle((int)mouse.X, (int)mouse.Y, 1, 1);
173  //Now we check if there is a collision
174  if (mouseColision.Intersects(_location))
175  {
176  //If the right button is pressed and the button wasn't clicked recently
177  if ((mouse.LeftButton == ButtonState.Pressed && _selector.IsTileSelected) /*&& !WasClicked*/)
178  {
179  ///Mouse location on texture
180  int mouseX = (mouse.X - _location.X) / 48;
181  int mouseY = (mouse.Y - _location.Y) / 48;
182 
183  switch (_currentLayer)
184  {
185  case (currentLayer.GROUND):
186  if (_selector.IsTileSelected)
187  {
188  _map.Ground.setItemAt(new Vector2(mouseX, mouseY), _selector.SelectedTile, _map.Player);
189  _map.CollisionLayer.setItemAt(new Vector2(mouseX, mouseY), _selector.SelectedTile.SourceRectangle, _map.Player);
190  }
191  break;
192  case (currentLayer.MASK):
193  layerLabel.Text = "MASK";
194  if (_selector.IsTileSelected)
195  _map.Mask.setItemAt(new Vector2(mouseX, mouseY), _selector.SelectedTile, _map.Player);
196  break;
197  case (currentLayer.FRINGE):
198  layerLabel.Text = "FRINGE";
199  if (_selector.IsTileSelected)
200  _map.Fringe.setItemAt(new Vector2(mouseX, mouseY), _selector.SelectedTile, _map.Player);
201  break;
202  }//endswitch
203  }//end leftbutton if
204 
205  //If the right button is pressed and the button wasn't clicked recently
206  if ((mouse.RightButton == ButtonState.Pressed) /*&& !WasClicked*/)

208  ///Mouse location on texture
